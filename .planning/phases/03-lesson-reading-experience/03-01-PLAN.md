       export interface LessonSections {
         article: string;
         vocabulary: string;
         expressions: string;
         comprehension: string;
       }

       export interface QABlock {
         question: string;
         answerEn: string;
         answerZh: string;
       }

       - parseSections(body: string): LessonSections — Split on ## 📖, ## 📚, ## 🔗, ## ❓ headers using regex. Each section captures content between its header and the next header (or end of string). Return trimmed strings. Empty input returns all empty strings.

       - extractSourceUrl(articleText: string): string | null — Match /^>\s*Source:\s*(https?:\/\/\S+)/m in the article text. Return the URL or null.

       - extractPreview(articleText: string, maxChars = 60): string — Split article text on newlines, filter to non-empty lines, skip lines matching /^\d{2}-\d{2}-\d{4}/ (timestamp format) and lines starting with > (blockquotes). Take the first remaining line. If longer than maxChars, truncate and append "...". Return empty string if no qualifying line found.

       - parseComprehension(text: string): QABlock[] — Match repeated **Q: {question}** / **A (EN):** {answer} / **A (中):** {answer} blocks. Use regex with g and s (dotAll) flags since answers can be multi-line. The pattern: /\*\*Q:\s*(.*?)\*\*\s*\n\*\*A \(EN\):\*\*\s*([\s\S]*?)\n\*\*A \(中\):\*\*\s*([\s\S]*?)(?=\n\*\*Q:|$)/g. Trim each captured group.

       - getTodayShanghai(): string — Use new Intl.DateTimeFormat('en-CA', { timeZone: 'Asia/Shanghai', year: 'numeric', month: '2-digit', day: '2-digit' }).format(new Date()) to produce a YYYY-MM-DD string.

       All functions are pure (no side effects, no mutation). Each returns a NEW object.

    import { getCollection } from 'astro:content';
    import Base from '../layouts/Base.astro';
    import { parseSections, extractSourceUrl, extractPreview, getTodayShanghai } from '../lib/lesson-parser';

    const entries = await getCollection('lessons');
    const lessons = entries.sort((a, b) => b.id.localeCompare(a.id));

    const todayId = getTodayShanghai();
    const hasTodayLesson = lessons.some(l => l.id === todayId);
    const featuredLesson = hasTodayLesson
      ? lessons.find(l => l.id === todayId)!
      : lessons[0];
    const showNoticeBar = !hasTodayLesson && lessons.length > 0;

    const recentLessons = lessons
      .filter(l => l.id !== featuredLesson?.id)
      .slice(0, 5);

    // Parse featured lesson sections
    const sections = featuredLesson ? parseSections(featuredLesson.body ?? '') : null;
    const sourceUrl = sections ? extractSourceUrl(sections.article) : null;

    // Build recent lesson previews
    const recentPreviews = recentLessons.map(l => {
      const s = parseSections(l.body ?? '');
      return { id: l.id, preview: extractPreview(s.article, 60) };
    });
