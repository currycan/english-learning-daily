       Collapse/expand (LESS-05, LESS-06, LESS-07):
       document.querySelectorAll('.section-block').forEach(function(block) {
         var btn = block.querySelector('.section-header');
         var body = block.querySelector('.section-body');
         var chevron = block.querySelector('.chevron');
         btn.addEventListener('click', function() {
           var collapsed = block.dataset.collapsed === 'true';
           body.style.gridTemplateRows = collapsed ? '1fr' : '0fr';
           chevron.style.transform = collapsed ? 'rotate(-180deg)' : '';
           block.dataset.collapsed = collapsed ? 'false' : 'true';
         });

       document.querySelectorAll('.reveal-btn').forEach(function(btn) {

           answer.classList.remove('hidden');
           btn.remove();
         });
       });

       localStorage read tracking (PROG-01 — Phase 04 dependency, per UI-SPEC):
       (function() {
         var lessonId = document.querySelector('h1').textContent.trim();
         try {
           var stored = JSON.parse(localStorage.getItem('read_lessons') || '[]');
           if (stored.indexOf(lessonId) === -1) {
             var updated = stored.concat([lessonId]);
             localStorage.setItem('read_lessons', JSON.stringify(updated));
           }
         } catch(e) {}
       })();
       Note: uses concat (immutable — creates new array) instead of push (mutates), per CLAUDE.md immutability constraint and UI-SPEC contract.
