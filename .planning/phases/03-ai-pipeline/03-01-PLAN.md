    """After Phase 3: main() reads Markdown from stdin and writes it to content path."""
    today = date(2026, 3, 23)
    fake_path = tmp_path / "2026-03-23.md"

    monkeypatch.setattr(cc, "get_beijing_date", lambda: today)
